<script>
    async function generatePDF() {
        const { jsPDF } = window.jspdf;
        const doc = new jsPDF();
        const now = new Date().toLocaleString();

        doc.setFontSize(18);
        doc.text("Raport Dzienny Elewatora Zbożowego", 14, 20);

        doc.setFontSize(10);
        doc.text(`Wygenerowano: ${now}`, 14, 28);
        doc.text(`Obsada: ${document.getElementById('staff').value || 'Nie podano'}`, 14, 34);

        doc.setFontSize(14);
        doc.text("1. Rejestr Czynności", 14, 45);

        doc.autoTable({
            startY: 50,
            head: [['Czas', 'Kategoria', 'Opis / Lokalizacja']],
            body: [
                ['08:15', 'Przyjęcie', 'Pszenica, 24t, nr rej. BIA 12345'],
                ['10:30', 'Przesyp', 'Z komory K-01 do K-05 (wietrzenie)'],
                ['12:00', 'Suszenie', 'Kukurydza, temp. zadana 90C']
            ],
            theme: 'grid'
        });

        let finalY = doc.lastAutoTable.finalY + 15;
        doc.text("2. Rejestr Awarii i Usterek", 14, finalY);

        const failureRows = Object.values(allFailures).map(item => [
            new Date(item.timestamp).toLocaleTimeString(),
            item.equipment,
            item.severity.toUpperCase(),
            item.description
        ]);

        doc.autoTable({
            startY: finalY + 5,
            head: [['Czas', 'Urządzenie', 'Priorytet', 'Opis usterki']],
            body: failureRows,
            headStyles: { fillColor: [231, 76, 60] },
            theme: 'striped'
        });

        const pageCount = doc.internal.getNumberOfPages();
        for (let i = 1; i <= pageCount; i++) {
            doc.setPage(i);
            doc.text(`Strona ${i} z ${pageCount}`, 180, 285);
        }

        doc.save(`Raport_Elewator_${new Date().toISOString().slice(0,10)}.pdf`);
    }
</script>
import { getAuth, signInWithEmailAndPassword, sendPasswordResetEmail, signOut } 
    from "https://www.gstatic.com/firebasejs/10.7.1/firebase-auth.js";

const auth = getAuth();

// LOGOWANIE
window.handleLogin = function() {
    const email = document.getElementById("login-email").value;
    const pass = document.getElementById("login-password").value;

    signInWithEmailAndPassword(auth, email, pass)
        .then(() => {
            document.getElementById("login-screen").style.display = "none";
            document.getElementById("app-content").style.display = "block";
        })
        .catch(err => {
            document.getElementById("login-error").innerText = "Błędny login lub hasło";
            document.getElementById("login-error").style.display = "block";
        });
};

// RESET HASŁA
window.resetPassword = function() {
    const email = document.getElementById("login-email").value;

    if (!email) {
        alert("Podaj adres e-mail, aby zresetować hasło.");
        return;
    }

    sendPasswordResetEmail(auth, email)
        .then(() => {
            alert("Wysłano link resetujący hasło na: " + email);
        })
        .catch(err => {
            alert("Błąd: " + err.message);
        });
};

// WYLOGOWANIE
window.handleLogout = function() {
    signOut(auth).then(() => {
        document.getElementById("login-screen").style.display = "block";
        document.getElementById("app-content").style.display = "none";
    });
};

<script type="module">
    import { getDatabase, ref, onValue, update } from "https://www.gstatic.com/firebasejs/10.7.1/firebase-database.js";

    const db = getDatabase();
    let allFailures = {};

    const failuresRef = ref(db, 'failures');
    onValue(failuresRef, (snapshot) => {
        allFailures = snapshot.val() || {};
        renderFailures();
    });

    window.renderFailures = function() {
        const filter = document.getElementById('filter_severity').value;
        const container = document.getElementById('failures_list');
        container.innerHTML = '';

        Object.keys(allFailures).reverse().forEach(id => {
            const item = allFailures[id];

            if (filter !== 'all' && item.severity !== filter) return;

            const card = document.createElement('div');
            card.className = 'card';
            card.style.borderLeft = `10px solid ${getSeverityColor(item.severity)}`;

            card.innerHTML = `
                <small>${new Date(item.timestamp).toLocaleString()}</small>
                <p><strong>Urządzenie:</strong> ${item.equipment}</p>
                <p><strong>Opis:</strong> ${item.description}</p>
                ${item.photo ? `<button onclick="showPhoto('${item.photo}')">Pokaż zdjęcie</button>` : ''}
                <div style="margin-top:10px;">
                    <button onclick="markAsFixed('${id}')" style="background:#2980b9; font-size: 0.8rem;">Oznacz jako Naprawione</button>
                </div>
            `;
            container.appendChild(card);
        });
    };

    function getSeverityColor(sev) {
        if (sev === 'critical') return '#e74c3c';
        if (sev === 'medium') return '#f1c40f';
        return '#2ecc71';
    }

    window.showPhoto = function(base64) {
        const win = window.open("");
        win.document.write(`<img src="${base64}" style="width:100%" />`);
    };

    window.markAsFixed = function(id) {
        const itemRef = ref(db, `failures/${id}`);
        update(itemRef, { is_fixed: true, fixed_at: new Date().toISOString() });
        alert("Zaktualizowano status awarii.");
    };

    // Prosta logika logowania (do uzupełnienia według Twojego systemu)
    window.handleLogin = function() {
        document.getElementById("login-screen").style.display = "none";
        document.getElementById("app-content").style.display = "block";
    };

    window.handleLogout = function() {
        document.getElementById("login-screen").style.display = "block";
        document.getElementById("app-content").style.display = "none";
    };
</script>
