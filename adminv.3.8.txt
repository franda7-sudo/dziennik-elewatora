            <div class="tech-field"><label>Producent</label><input type="text" id="edit-prod"></div>

<script type="module">
    import { initializeApp } from "https://www.gstatic.com/firebasejs/10.7.1/firebase-app.js";
    import { getDatabase, ref, onValue, set, remove } from "https://www.gstatic.com/firebasejs/10.7.1/firebase-database.js";

    const firebaseConfig = {
        apiKey: "AIzaSyDerQvG4mlaa-v2ZP4Sowx6_P9XkVfpluM",
        authDomain: "elewator-dziennik.firebaseapp.com",
        databaseURL: "https://elewator-dziennik-default-rtdb.europe-west1.firebasedatabase.app",
        projectId: "elewator-dziennik",
        storageBucket: "elewator-dziennik.firebasestorage.app",
        messagingSenderId: "896699001772",
        appId: "1:896699001772:web:07633e4f9f32b5f9ad6809"
    };

    const app = initializeApp(firebaseConfig);
    const db = getDatabase(app);
    let activeId = null;

    onValue(ref(db, 'settings/machines'), snap => {
        const data = snap.val() || {};
        document.getElementById('machines-list').innerHTML = Object.keys(data).map(id => `
            <div class="machine-item" onclick="openCard('${id}', '${data[id].name}')">
                <strong>${id} - ${data[id].name}</strong>
                <button onclick="event.stopPropagation(); removeMach('${id}')" style="color:red; background:none;">usuń</button>
            </div>
        `).join('');
    });

    onValue(ref(db, 'settings/workers'), snap => {
        const list = snap.val() || [];
        document.getElementById('workers-list').innerHTML = list.map((w,i) => `<div>${w} <button onclick="removeItem('workers', ${i})">x</button></div>`).join('');
    });

    window.openCard = (id, name) => {
        activeId = id;
        document.getElementById('modal-title').innerText = id + " - " + name;
        onValue(ref(db, `settings/machines/${id}`), s => {
            const m = s.val() || {};
            document.getElementById('edit-prod').value = m.prod || '';
            document.getElementById('edit-power').value = m.power || '';
            document.getElementById('edit-bear').value = m.bear || '';
            document.getElementById('edit-dtr').value = m.dtr || '';
            document.getElementById('edit-notes').value = m.notes || '';
        }, {onlyOnce: true});
        
        onValue(ref(db, 'failures'), snap => {
            const all = snap.val() || {};
            document.getElementById('history-body').innerHTML = Object.values(all).filter(f => f.machine === id).reverse().map(f => `
                <tr><td>${f.timestamp.split('T')[0]}</td><td>${f.type}</td><td>${f.description}</td></tr>
            `).join('');
        }, {onlyOnce: true});
        document.getElementById('machineModal').style.display='block';
    };

    window.saveDetails = () => {
        const details = {
            prod: document.getElementById('edit-prod').value,
            power: document.getElementById('edit-power').value,
            bear: document.getElementById('edit-bear').value,
            dtr: document.getElementById('edit-dtr').value,
            notes: document.getElementById('edit-notes').value
        };
        onValue(ref(db, `settings/machines/${activeId}/name`), n => {
            details.name = n.val();
            set(ref(db, `settings/machines/${activeId}`), details).then(() => alert("Zapisano!"));
        }, {onlyOnce: true});
    };

    window.addMachine = () => {
        const id = document.getElementById('m-id').value.toUpperCase();
        const name = document.getElementById('m-name').value;
        if(id && name) set(ref(db, `settings/machines/${id}`), { name: name });
    };

    window.removeMach = (id) => { if(confirm("Usunąć?")) remove(ref(db, `settings/machines/${id}`)); };
    
    window.addItem = (key, inputId) => {
        const val = document.getElementById(inputId).value;
        onValue(ref(db, 'settings/'+key), s => {
            const l = s.val() || []; l.push(val);
            set(ref(db, 'settings/'+key), l);
        }, {onlyOnce: true});
    };

    window.removeItem = (key, i) => {
        onValue(ref(db, 'settings/'+key), s => {
            const l = s.val() || []; l.splice(i,1);
            set(ref(db, 'settings/'+key), l);
        }, {onlyOnce: true});
    };

    window.switchTab = (id, b) => {
        document.querySelectorAll('.section').forEach(s => s.classList.remove('active'));
        document.getElementById('section-'+id).classList.add('active');
        document.querySelectorAll('.tab-btn').forEach(btn => btn.classList.remove('active'));
        b.classList.add('active');
    };
</script>
